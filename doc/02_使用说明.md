## 快速开始

### 最简单的策略

```python
from datetime import datetime
import backtrader as bt

# 定义策略
class SmaCross(bt.SignalStrategy):
    def __init__(self):
        sma1 = bt.ind.SMA(period=10)
        sma2 = bt.ind.SMA(period=30)
        crossover = bt.ind.CrossOver(sma1, sma2)
        self.signal_add(bt.SIGNAL_LONG, crossover)

# 创建引擎
cerebro = bt.Cerebro()

# 添加策略
cerebro.addstrategy(SmaCross)

# 添加数据
data = bt.feeds.YahooFinanceData(dataname='MSFT',
                                 fromdate=datetime(2020, 1, 1),
                                 todate=datetime(2021, 12, 31))
cerebro.adddata(data)

# 执行回测
cerebro.run()

# 绘制结果
cerebro.plot()
```

## 核心概念

### 1. Cerebro（引擎）

Cerebro是Backtrader的核心控制中心，负责协调所有组件。

```python
import backtrader as bt

# 创建Cerebro实例
cerebro = bt.Cerebro()

# 配置参数
cerebro = bt.Cerebro(
    preload=True,      # 预加载数据
    runonce=True,      # 向量化运行模式
    stdstats=True,     # 添加默认观察者
    cash=10000,        # 初始资金
    commission=0.001   # 默认佣金
)
```

**主要方法**:

| 方法 | 说明 |
|------|------|
| `adddata(data)` | 添加数据源 |
| `addstrategy(strategy, **kwargs)` | 添加策略 |
| `addanalyzer(analyzer, **kwargs)` | 添加分析器 |
| `addobserver(observer, **kwargs)` | 添加观察者 |
| `addwriter(writer, **kwargs)` | 添加数据输出器 |
| `addsizer(sizer, **kwargs)` | 添加仓位管理器 |
| `run(**kwargs)` | 执行回测 |
| `plot(**kwargs)` | 绘制结果 |

### 2. Strategy（策略）

策略是交易逻辑的核心，需要继承`bt.Strategy`类。

```python
class MyStrategy(bt.Strategy):
    params = (
        ('period', 20),
        ('printlog', True),
    )

    def __init__(self):
        # 初始化指标
        self.sma = bt.ind.SMA(self.data.close, period=self.p.period)
        self.rsi = bt.ind.RSI(self.data.close)

    def next(self):
        # 每个bar执行一次
        if self.rsi < 30:
            self.buy()
        elif self.rsi > 70:
            self.sell()

    def notify_order(self, order):
        # 订单状态通知
        if order.status in [order.Completed]:
            if self.p.printlog:
                print(f'订单执行: {order.executed.price}')

    def notify_trade(self, trade):
        # 交易通知
        if trade.isclosed:
            print(f'交易盈亏: {trade.pnl}')

    def log(self, txt, dt=None):
        # 自定义日志方法
        dt = dt or self.datas[0].datetime.date(0)
        print(f'{dt.isoformat()}, {txt}')
```

**生命周期方法**:

| 方法 | 说明 |
|------|------|
| `__init__()` | 初始化，创建指标 |
| `prenext()` | 数据不足时调用 |
| `nextstart()` | 第一次调用next前执行 |
| `next()` | 每个bar执行一次，核心逻辑 |
| `notify_order(order)` | 订单状态变化通知 |
| `notify_trade(trade)` | 交易完成通知 |
| `notify_cashvalue(cash, value)` | 现金和价值通知 |
| `notify_fund(cash, value, fundvalue)` | 基金价值通知 |

**交易方法**:

```python
# 买入
order = self.buy(data=None, size=None, price=None, limit=None,

# 卖出
```

### 4. Indicators（指标）

#### 内置指标使用

```python
class MyStrategy(bt.Strategy):
    def __init__(self):
        # 移动平均
        self.sma = bt.ind.SMA(period=20)
        self.ema = bt.ind.EMA(period=20)

        # RSI
        self.rsi = bt.ind.RSI(period=14)

        # MACD
        self.macd = bt.ind.MACD()

        # 布林带
        self.boll = bt.ind.BollingerBands(period=20, devfactor=2)

        # ATR
        self.atr = bt.ind.ATR(period=14)

        # 交叉指标
        self.crossover = bt.ind.CrossOver(self.sma, self.data.close)
```

#### 自定义指标

```python
class MyIndicator(bt.Indicator):
    lines = ('signal',)  # 定义输出线

    params = (
        ('period', 20),
    )

    def __init__(self):
        self.lines.signal = bt.ind.SMA(self.data.close, period=self.p.period)
```

### 4. 订单类型

```python
class OrderStrategy(bt.Strategy):
    def next(self):
        # 市价单
        self.buy(exectype=bt.Order.Market)

        # 限价单
        self.buy(exectype=bt.Order.Limit, price=self.data.close[0] * 0.99)

        # 止损单
        self.buy(exectype=bt.Order.Stop, price=self.data.close[0] * 1.01)

        # 止损限价单
        self.buy(exectype=bt.Order.StopLimit,
                 price=self.data.close[0] * 1.01,  # 触发价
                 limit=self.data.close[0] * 1.02)   # 限价

        # 追踪止损
        self.buy(exectype=bt.Order.StopTrail,
                 trailamount=self.data.close[0] * 0.02)  # 追踪距离

        # OCO订单（One-Cancels-Other）
        o1 = self.buy(price=self.data.close[0] * 0.99)
        o2 = self.sell(price=self.data.close[0] * 1.01, oco=o1)

        # Bracket订单（ bracket订单）
        main = self.buy(price=self.data.close[0])
        self.sell(price=self.data.close[0] * 1.05, parent=main)  # 止盈
        self.sell(price=self.data.close[0] * 0.95, parent=main)  # 止损
```

```python
def next(self):
    # 获取持仓
    position = self.getposition()

    # 持仓大小
    size = position.size

    # 持仓均价
    price = position.price

    # 是否有持仓

    # 平仓
```

### 3. 订单管理

```python
def notify_order(self, order):
    if order.status == order.Created:
        print('订单已创建')

    elif order.status == order.Submitted:
        print('订单已提交')

    elif order.status == order.Accepted:
        print('订单已接受')

    elif order.status == order.Partial:
        print('订单部分成交')

    elif order.status == order.Completed:
        print(f'订单完全成交: 价格={order.executed.price}, '
              f'数量={order.executed.size}, '
              f'成本={order.executed.value}')

    elif order.status == order.Canceled:
        print('订单已取消')

    elif order.status == order.Rejected:
        print('订单被拒绝')

    elif order.status == order.Margin:
        print('订单保证金不足')
```
