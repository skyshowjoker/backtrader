### 1. 自定义策略开发

#### 策略模板

```python
import backtrader as bt

class CustomStrategy(bt.Strategy):
    """自定义策略模板"""

    # 策略参数
    params = (
        ('param1', 10),      # 参数1，默认值10
        ('param2', 20),      # 参数2，默认值20
        ('printlog', True),  # 是否打印日志
    )

    def __init__(self):
        """初始化：创建指标和辅助变量"""
        # 创建指标
        self.indicator1 = bt.ind.SMA(period=self.p.param1)
        self.indicator2 = bt.ind.SMA(period=self.p.param2)

        # 创建信号
        self.signal = bt.ind.CrossOver(self.indicator1, self.indicator2)

        # 订单变量
        self.order = None
        self.buyprice = None
        self.buycomm = None

    def prenext(self):
        """数据不足时调用"""
        pass

    def nextstart(self):
        """第一次next前调用"""
        if self.p.printlog:
            print('策略开始执行')

    def next(self):
        """每个bar执行一次：核心交易逻辑"""
        # 检查是否有未完成订单
        if self.order:
            return

        # 检查持仓
        if not self.position:
            # 无持仓，检查买入信号
            if self.signal > 0:
                self.log('买入信号')
                self.order = self.buy()

        else:
            # 有持仓，检查卖出信号
            if self.signal < 0:
                self.log('卖出信号')
                self.order = self.sell()

    def notify_order(self, order):
        """订单状态通知"""
        if order.status in [order.Submitted, order.Accepted]:
            return

        if order.status in [order.Completed]:
            if order.isbuy():
                self.log(f'买入执行: 价格={order.executed.price:.2f}, '
                        f'成本={order.executed.value:.2f}, '
                        f'手续费={order.executed.comm:.2f}')
                self.buyprice = order.executed.price
                self.buycomm = order.executed.comm
            else:
                self.log(f'卖出执行: 价格={order.executed.price:.2f}, '
                        f'成本={order.executed.value:.2f}, '
                        f'手续费={order.executed.comm:.2f}')

        elif order.status in [order.Canceled, order.Margin, order.Rejected]:
            self.log('订单取消/保证金不足/拒绝')

        self.order = None

    def notify_trade(self, trade):
        """交易完成通知"""
        if not trade.isclosed:
            return

        self.log(f'交易盈亏: 毛利={trade.pnl:.2f}, 净利={trade.pnlcomm:.2f}')

    def notify_cashvalue(self, cash, value):
        """现金和价值通知"""
        if self.p.printlog:
            print(f'现金: {cash:.2f}, 总价值: {value:.2f}')

    def log(self, txt, dt=None):
        """日志函数"""
        dt = dt or self.datas[0].datetime.date(0)
        print(f'{dt.isoformat()}, {txt}')
```

class MultiDataStrategy(bt.Strategy):
    """多数据源策略"""

    params = (
        ('period', 20),
    )

    def __init__(self):
        # 为每个数据源创建指标
        self.sma_list = []
        for data in self.datas:
            sma = bt.ind.SMA(data.close, period=self.p.period)
            self.sma_list.append(sma)

    def next(self):
        for i, data in enumerate(self.datas):
            pos = self.getposition(data)
            sma = self.sma_list[i]

            if not pos:
                if data.close[0] > sma[0]:
                    self.buy(data=data)
            else:
                if data.close[0] < sma[0]:
                    self.sell(data=data)

class ComplexIndicator(bt.Indicator):
    """复杂自定义指标"""

    lines = ('upper', 'middle', 'lower',)
    params = (
        ('period', 20),
        ('devfactor', 2.0),
    )

    # 绘图设置
    plotinfo = dict(
        subplot=False,  # 与主图重叠
        plotname='MyBB',
    )

    plotlines = dict(
        upper=dict(color='red'),
        middle=dict(color='blue'),
        lower=dict(color='green'),
    )

    def __init__(self):
        # 中轨：移动平均
        self.lines.middle = bt.ind.SMA(self.data.close, period=self.p.period)

        # 标准差
        std = bt.ind.StdDev(self.data.close, period=self.p.period)

        # 上轨和下轨
        self.lines.upper = self.lines.middle + std * self.p.devfactor
        self.lines.lower = self.lines.middle - std * self.p.devfactor

    # 或者使用next方法手动计算
    def next(self):
        # 手动计算逻辑
        data = self.data.close.get(size=self.p.period)
        if len(data) >= self.p.period:
            mean = sum(data) / self.p.period
            std = (sum((x - mean) ** 2 for x in data) / self.p.period) ** 0.5

            self.lines.middle[0] = mean
            self.lines.upper[0] = mean + std * self.p.devfactor
            self.lines.lower[0] = mean - std * self.p.devfactor
```

class CustomAnalyzer(bt.Analyzer):
    """自定义分析器"""

    params = (
        ('period', 30),
    )

    def __init__(self):
        # 初始化变量
        self.trades_count = 0
        self.wins = 0
        self.losses = 0
        self.total_pnl = 0

    def notify_trade(self, trade):
        """交易完成通知"""
        if trade.isclosed:
            self.trades_count += 1
            self.total_pnl += trade.pnlcomm

            if trade.pnlcomm > 0:
                self.wins += 1
            else:
                self.losses += 1

    def get_analysis(self):
        """返回分析结果"""
        win_rate = self.wins / self.trades_count if self.trades_count > 0 else 0

        return {
            'total_trades': self.trades_count,
            'wins': self.wins,
            'losses': self.losses,
            'win_rate': win_rate,
            'total_pnl': self.total_pnl,
            'avg_pnl': self.total_pnl / self.trades_count if self.trades_count > 0 else 0,
        }
```

class CustomSizer(bt.Sizer):
    """自定义仓位管理器"""

    params = (
        ('risk_per_trade', 0.02),  # 每笔交易风险2%
        ('stop_loss_pct', 0.05),   # 止损5%
    )

    def _getsizing(self, comminfo, cash, data, isbuy):
        """计算仓位大小"""
        # 获取当前价格
        price = data.close[0]

        # 计算止损金额
        stop_loss_amount = price * self.p.stop_loss_pct

        # 计算可承受的最大损失
        max_loss = cash * self.p.risk_per_trade

        # 计算股数
        size = int(max_loss / stop_loss_amount)

        # 返回仓位大小
        return size

# 注意：某些自定义指标可能不支持向量化

class VectorizedIndicator(bt.Indicator):
    lines = ('signal',)

    def next(self):
        # 逐个bar计算
        self.lines.signal[0] = self.data.close[0] * 0.1

    def once(self, start, end):
        # 向量化计算
        for i in range(start, end):
            self.lines.signal[i] = self.data.close[i] * 0.1

### 2. 风险管理

```python
class RiskManagedStrategy(bt.Strategy):
    params = (
        ('max_position_pct', 0.20),  # 单个持仓不超过20%
        ('max_drawdown_pct', 0.15),  # 最大回撤15%
    )

    def __init__(self):
        self.initial_value = self.broker.getvalue()
        self.peak_value = self.initial_value

    def next(self):
        # 更新峰值
        current_value = self.broker.getvalue()
        self.peak_value = max(self.peak_value, current_value)

        # 检查回撤
        drawdown = (self.peak_value - current_value) / self.peak_value
        if drawdown > self.p.max_drawdown_pct:
            print(f'触发最大回撤限制: {drawdown*100:.2f}%')
            self.close()  # 平掉所有仓位
            return

        # 检查仓位集中度
        if self.position:
            position_value = self.position.size * self.data.close[0]
            position_pct = position_value / current_value
            if position_pct > self.p.max_position_pct:
                print(f'仓位过大: {position_pct*100:.2f}%')
                # 减仓
                reduce_size = int(self.position.size * 0.5)
                self.sell(size=reduce_size)
```

import smtplib
from email.mime.text import MIMEText

class MonitoredStrategy(bt.Strategy):
    def send_alert(self, subject, message):
        """发送报警邮件"""
        msg = MIMEText(message)
        msg['Subject'] = subject
        msg['From'] = 'your_email@gmail.com'
        msg['To'] = 'receiver@gmail.com'

        # 发送邮件
        with smtplib.SMTP('smtp.gmail.com', 587) as server:
            server.starttls()
            server.login('your_email@gmail.com', 'password')
            server.send_message(msg)

    def notify_trade(self, trade):
        """交易完成时发送通知"""
        if trade.isclosed:
            pnl = trade.pnlcomm
            subject = f'交易完成 - {"盈利" if pnl > 0 else "亏损"}'
            message = f'盈亏: {pnl:.2f}'
            self.send_alert(subject, message)
